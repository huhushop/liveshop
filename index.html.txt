<script>
    let currentQty = 0;
    const unitPrice = 680;

    // 1. 初始化 LIFF
    async function initLiff() {
        try {
            await liff.init({ liffId: "你的_LIFF_ID_填在這裡" });
            if (!liff.isLoggedIn()) {
                liff.login();
            }
        } catch (err) {
            console.error("LIFF 初始化失敗", err);
        }
    }
    initLiff();

    // 2. 數量控制
    function changeQty(num) {
        currentQty = Math.max(0, currentQty + num);
        document.getElementById('qty-val').innerText = currentQty;
        document.getElementById('total-price').innerText = `NT$ ${currentQty * unitPrice}`;
    }

    // 3. 提交訂單
    async function submitOrder() {
        if (currentQty === 0) return alert("請先選擇數量");

        const profile = await liff.getProfile();
        const note = document.getElementById('note-1').value;

        const orderData = {
            userName: profile.displayName,
            itemName: "日系質感手感咖啡杯",
            quantity: currentQty,
            note: note
        };

        // A. 傳送到 Google Sheets (這裡填你部署後的 GAS 網址)
        fetch('你的_GAS_部署網址', {
            method: 'POST',
            mode: 'no-cors', // 重要：避免跨網域問題
            body: JSON.stringify(orderData)
        });

        // B. 發送訊息回群組
        if (liff.isInClient()) {
            await liff.sendMessages([{
                type: 'text',
                text: `🛒 【HuHuShop 下單明細】\n──────────────\n買家：${orderData.userName}\n品項：${orderData.itemName}\n數量：${orderData.quantity}\n備註：${orderData.note || '無'}\n──────────────\n已成功加入 HuHu 的採買清單！`
            }]);
            liff.closeWindow();
        } else {
            alert("訂單已送出（非 LINE 環境僅模擬）");
        }
    }
</script>
